create table country (
country_id integer,
country_name varchar(250),
primary key (country_id)
);

create table kk (
countryid integer references country(country_id),
male_2010 real,
male_2014 real,
primary key (countryid)
);

create table kvk (
countryid integer references country(country_id),
female_2010 real,
female_2014 real,
primary key (countryid)
);

create table kk_pr (
countryid integer references country(country_id),
male_pr_2010 real,
male_pr_2014 real,
primary key (countryid)
);

create table kvk_pr (
countryid integer references country(country_id),
female_pr_2010 real,
female_pr_2014 real,
primary key (countryid)
);
